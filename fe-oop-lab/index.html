Encapsulation


const base = 1000;
const overtime = 10;
const rate = 20;

function getWage(base, overtime, rate) {
    return base + overtime * rate;
}
      

const employee  = {
    base: 1000,
    overtime: 10,
    rate: 20,
    getWage: function() {
        return this.base + (this.overtime * this.rate);
    }
}

employee.getWage();
The best functions have as few arguments as possible.
